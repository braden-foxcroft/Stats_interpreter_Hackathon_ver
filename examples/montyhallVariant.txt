

# A variant of the monty hall problem. In this variant, the host doesn't know which door has the prize, so he randomly opens one.
# By chance, the door he opens is empty. The question is the same: what are the odds that you win if you switch? In this variant,
# switching is actually inconsequential.

set doors = 1,2,3
select prize from doors
select playerChoice from doors

select host from doors where host != playerChoice
bychance host != prize

select switch from doors where switch != playerChoice and switch != host

if switch == prize
	pass
fail
